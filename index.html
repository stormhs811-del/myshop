# Report Card Program for 5 Students for i in range(1, 6): print(f"\nStudent {i}") name = input("Enter student name: ") # Input marks math = int(input("Math: ")) science = int(input("Science: ")) english = int(input("English: ")) ict = int(input("ICT: ")) social = int(input("Social Studies: ")) # Calculate total and average total = math + science + english + ict + social average = total / 5 # Assign grade if average >= 90: grade = "A" elif average >= 75: grade = "B" elif average >= 60: grade = "C" elif average >= 50: grade = "D" else: grade = "F" # Output print("\n--- Report Card ---") print("Name:", name) print("Total:", total) print("Average:", average) print("Grade:", grade)
